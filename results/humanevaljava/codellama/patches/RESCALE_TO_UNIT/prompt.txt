    public static double[] rescale_to_unit(double[] numbers){
        double min = Double.MAX_VALUE;
        double max = Double.MIN_VALUE;
        for (int i = 0; i < numbers.length; i += 1) {

                min = numbers[i];
            }
        }
        double[] result = new double[numbers.length];
